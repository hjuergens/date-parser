23 trait CardinalParsers extends RegexParsers {
24   private def numerical : Parser[Cardinal] =
25     """(\d+)""".r ^^
26       { Cardinal.fromString }
27 
28   private def textual : Parser[Cardinal] =
29     ( "one" | "two" | "three" | "four" | "five" | "six" | "seven" | "eight" | "nine" | "ten" | "eleven" | "twelve") ^^
30       { Cardinal.string2Long.andThen(Cardinal.apply) }
31 
32   def cardinal : Parser[Cardinal] = textual | numerical
33 }

35 trait OrdinalParsers  extends RegexParsers with ExtendedRegexParsers {
36   val regExp = """(\d*1)st|(\d*2)nd|(\d*3)rd|(\d*[4,5,6,7,8,9,0])th|(1[1,2,3])th""".r
37 
38   private def textual = ("first" | "eleventh" | "second" | "twelfth" | "third" | "thirteenth"| "fourth" |
39     "fourteenth" | "fifth" | "fifteenth" | "sixth" | "sixteenth" | "seventh " | "seventeenth" |
40     "eighth" | "eighteenth" | "ninth" | "nineteenth" | "tenth" | "twentieth") ^^
41     {
42       Ordinal.string2Int.andThen(Ordinal.apply)
43     }
44 
45   private def ordinalNumbers : Parser[Ordinal] = RegexParser(regExp) ^^
46     { case m =>
47       val matchingStr = (1 to m.groupCount).map(m.group).filterNot(_ == null).head
48       Ordinal(Integer.parseInt(matchingStr))
49     }
50 
51   private def numerical :  Parser[Ordinal] = RegexParser("""(\d+)\.""".r) ^^
52     { case m => Ordinal(Integer.parseInt(m.group(1))) }
53 
54 
55   assert( parseAll(ordinalNumbers,
56     "1st 2nd 3rd 4th 5th 6th 7th 8th 9th 10th 11th 12th 13th 14th 15th 16th 17th 18th 19th 20th 21st 22nd 23rd 24th 25th 26th 27th 28th 29th 30th 101st 102nd 103rd 104th 105th 106th 107th 108th 109th 110th").isEmpty
57   )
58 
59   def ordinal : Parser[Ordinal] = textual | numerical | ordinalNumbers
60 }
