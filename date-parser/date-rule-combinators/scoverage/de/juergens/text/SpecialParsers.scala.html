75   def season(dayInMonth:Int, month:Int) : String =
76   {
77     //The condition when the season start
78     if (((month == 12)&&(dayInMonth>=16))||((month<=3)&&(dayInMonth<=15))){
79       "Winter"
80     }
81     else if (((month >= 3)&&(dayInMonth>=16))||((month<=6)&&(dayInMonth<=15))){
82       "Spring"
83     }
84     else if (((month >= 6)&&(dayInMonth>=16))||((month<=9)&&(dayInMonth<=15))){
85       "Summer"
86     }
87     else if (((month >= 9)&&(dayInMonth>=16))||((month<=12)&&(dayInMonth<=15))){
88       "Fall"
89     }
90     else "fail"
91   }
