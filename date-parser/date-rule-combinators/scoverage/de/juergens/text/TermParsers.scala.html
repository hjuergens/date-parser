22   def standardPeriod: Parser[Period] = wholeNumber ~ timeUnitTerm ^^ {
23     case number ~ ChronoUnit.DAYS => Period.ofDays(number.toInt)
24     case number ~ ChronoUnit.WEEKS => Period.ofWeeks(number.toInt)
25     case number ~ ChronoUnit.MONTHS => Period.ofMonths(number.toInt)
26     case number ~ ChronoUnit.YEARS => Period.ofYears(number.toInt)
27   }

29   private def infinityPeriod: Parser[Period] = "Infinity" ^^
30     { _ => Period.ofYears(Integer.MAX_VALUE) }

32   private def lPeriod: Parser[Period] = "L" ^^
33     { _ => Period.ofWeeks(3) }
34 
35   private def periodSum: Parser[Period] = (standardPeriod *) ^^
36     { case periods => periods.foldRight(Period.ZERO)((p, s) => p plus s) }

38   def periodTerm: Parser[Period] = infinityPeriod | lPeriod | periodSum

122   object nextQuarterMonth extends TemporalAdjuster {
123     override def adjustInto(temporal: Temporal): Temporal = {
124       val currentMonthValue = Month.from(temporal).getValue
125       val nm = quaterlies.find(currentMonthValue < _.getValue)
126       if(nm.isDefined) {
127         temporal.`with`(nm.get)
128       } else {
129         assert(Month.from(temporal) == Month.DECEMBER)
130         temporal.plus(1, ChronoUnit.YEARS)
131         temporal.`with`(quaterlies.head)
132       }
133     }

136   object nextQuarterImm extends TemporalAdjuster {
137     override def adjustInto(temporal: Temporal): Temporal = {
138       val currentDayOfMonth = DayOfMonth.from(temporal)
139 
140       val thirdWednesdayInCurrentMonth = temporal.`with`(thirdWednesday)
141 
142       if(quaterlies.contains( Month.from(temporal) )) {
143         if (currentDayOfMonth.getValue < DayOfMonth.from(thirdWednesdayInCurrentMonth).getValue) {
144           temporal.`with`(thirdWednesday)
145         } else {
146           temporal.`with`(nextQuarterMonth).`with`(thirdWednesday)
147         }
148       } else {
149         temporal.`with`(nextQuarterMonth).`with`(thirdWednesday)
150       }
151     }
