27    * Returns the Julian day number that begins at noon of
28    * this day, Positive year signifies A.D., negative year B.C.
29    * Remember that the year after 1 B.C. was 1 A.D.
30    *
31    * ref :
32    *  Numerical Recipes in C, 2nd ed., Cambridge University Press 1992
33    */
34   // Gregorian Calendar adopted Oct. 15, 1582 (2299161)
35   val JGREG :Int = 15 + 31*(10+12*1582)
36   val HALFSECOND:Double = 0.5
37 
38   def toJulian(date:ZonedDateTime):Double = {
39     require(date.getZone.getId equals "UT")
40     ZonedDateTime.now(ZoneId.of("UT"))
41     val year= date.getYear
42     val month = date.getMonthValue // jan=1, feb=2,...
43     val day = date.getDayOfMonth
44     var julianYear :Int= year
45     if (year < 0) julianYear+=1
46     var julianMonth = month
47     if (month > 2) {
48       julianMonth+=1
49     }
50     else {
51       julianYear-=1
52       julianMonth += 13
53     }
54 
55     var julian :Double = (java.lang.Math.floor(365.25 * julianYear)
56       + java.lang.Math.floor(30.6001*julianMonth) + day + 1720995.0)
57     if (day + 31 * (month + 12 * year) >= JGREG) {
58       // change over to Gregorian calendar
59       val ja:Int = (0.01 * julianYear).toInt
60       julian += 2 - ja + (0.25 * ja)
61     }
62     java.lang.Math.floor(julian)
63   }

65   /**
66    * Converts a Julian day to a calendar date
67    * ref :
68    * Numerical Recipes in C, 2nd ed., Cambridge University Press 1992
69    */
70   def fromJulian(inJulian:Double) : LocalDate ={
71     val julian :Double = inJulian + HALFSECOND / 86400.0
72     var ja :Int = julian.toInt
73     if (ja >= JGREG) {
74       val jalpha = (((ja - 1867216) - 0.25) / 36524.25).toInt
75       ja = ja + 1 + jalpha - jalpha / 4
76     }
77 
78     val jb :Int = ja + 1524
79     val jc = (6680.0 + ((jb - 2439870) - 122.1) / 365.25).toInt
80     val jd :Int = 365 * jc + jc / 4
81     val je = ((jb - jd) / 30.6001).toInt
82     val day :Int = jb - jd - (30.6001 * je).toInt
83     var month :Int = je - 1
84     if (month > 12) month = month - 12
85     var year = jc - 4715
86     if (month > 2) year-=1
87     if (year <= 0) year-=1
88 
89     LocalDate.of(year, month, day)
90   }

92     def dateToJulian(date : java.util.Calendar) : Double ={
93         val year = date.get(java.util.Calendar.YEAR)
94         val month = date.get(java.util.Calendar.MONTH) + 1
95         val day = date.get(java.util.Calendar.DAY_OF_MONTH)
96         val hour = date.get(java.util.Calendar.HOUR_OF_DAY)
97         val minute = date.get(java.util.Calendar.MINUTE)
98         val second = date.get(java.util.Calendar.SECOND)
99 
100         val extra = (100.0 * year) + month - 190002.5
101         (367.0 * year) -
102                 (Math.floor(7.0 * (year + Math.floor((month + 9.0) / 12.0)) / 4.0)) +
103                 Math.floor((275.0 * month) / 9.0) +
104                 day + ((hour + ((minute + (second / 60.0)) / 60.0)) / 24.0) +
105                 1721013.5 - ((0.5 * extra) / Math.abs(extra)) + 0.5
106     }

108       def currentDateToJulianDate(cal:GregorianCalendar): Double = {
109           var jd = 0.0
110               val hour24 = cal.get(java.util.Calendar.HOUR_OF_DAY)   // 0..23
111               val min = cal.get(java.util.Calendar.MINUTE)  //0..59
112               val sec = cal.get(java.util.Calendar.SECOND)
113               val day = cal.get(java.util.Calendar.DAY_OF_MONTH)
114               val month = cal.get(java.util.Calendar.MONTH) + 1
115               val year = cal.get(java.util.Calendar.YEAR)
116 
117               jd = (367 * year) - (7 * (year + ((month + 9) / 12).toInt) / 4).toInt -
118                 ((3 * ((year + (month - 9) / 7) / 100).toInt + 1) / 4).toInt
119                       + ((275 * month) / 9).toInt + day + 1721028.5
120                       + (hour24 + (min / 60) + (sec / 3600)) / 24
121           jd
122       }

163 /*
164 There is a lot of variation around the idea of a "Julian date". You can have the Modified Julian Date (JD) or the Truncated Julian Date (TJD). The main difference is the starting for counting the days.
165 
166 Before Y2K, many applications (especially mainframe systems) were storing dates in a format called "the Julian format". This format is a 5 digit number, consisting of a 2 digit year and a 3 digit day-of-year number. For example, 17-July-1998 is stored as 98221, since 17-July is the 221th day of the year. This format is not really useful since Y2K! The main reason for using the 5-digits Julian date was to save disk space and still have a format easy to use to handle. dates.
167 
168 A variation of this idea is to used the first four digits for the year and 3 digits for day-of-year, 17-July-1998 will be represented by 1998221.
169 
170 This is a very simple format which can be manipulated easily from Java.
171 */
