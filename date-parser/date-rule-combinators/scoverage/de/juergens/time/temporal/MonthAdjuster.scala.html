9 case class MonthAdjuster(ordinal: Ordinal, month: Month, direction: Direction)
10   extends LocalDateAdjuster {
11 
12   private def signum = direction
13   private val Ordinal(number) = ordinal

15   val isBeforeMonth = (accessor: TemporalAccessor) => accessor.get(ChronoField.MONTH_OF_YEAR) < month.getValue
16   val isAfterMonth = (accessor: TemporalAccessor) => accessor.get(ChronoField.MONTH_OF_YEAR) > month.getValue
17   val isMonth = (accessor: TemporalAccessor) => accessor.get(ChronoField.MONTH_OF_YEAR) == month.getValue
18 
19   override def adjustInto(temporal: Temporal): Temporal = {
20     var date = temporal
21 
22     require(temporal.isSupported(ChronoField.MONTH_OF_YEAR), s"$temporal has to support month-of-year")
23     temporal.ensuring(_.isSupported(ChronoField.MONTH_OF_YEAR))
24 
25     if (number > 0)
26       date = date.`with`(ChronoField.MONTH_OF_YEAR, month.getValue)
27 
28     val rest: Int = {
29       if (isMonth(date)) {
30         signum * number
31       } else {
32         if (direction == Up && isBeforeMonth(date)) signum * number - 1
33         else if (direction == Down && isAfterMonth(date)) signum * number + 1
34         else signum * number
35       }
36     }
37 
38     val currentYear = date.get(ChronoField.YEAR)
39     date = date.`with`(ChronoField.YEAR, currentYear + rest)
40 
41     direction match {
42       case Up => date = TemporalAdjusters.lastDayOfMonth().adjustInto(date)
43       case Down => date = TemporalAdjusters.firstDayOfMonth().adjustInto(date)
44     }
45 
46     assert( temporal.get(ChronoField.MONTH_OF_YEAR) == month.getValue )
47 
48     date
49   }
