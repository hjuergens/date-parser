191 case class IntervalImpl[T](lower: LowerBorder[T], upper: UpperBorder[T])
192                           (implicit ord : scala.math.Ordering[T])
193   extends Interval[T] {
194 
195   type I = IntervalImpl[T]
196 
197   override def signum(implicit num:Numeric[T]) :T = num.minus(upper.rhs,lower.lhs)
198 
199   /**union */
200   def +(that: IntervalImpl[T]) : IntervalImpl[T] = {
201     if(this.upper.rhs > that.lower.lhs && this.lower.lhs < that.upper.rhs)
202       new IntervalImpl[T](this.lower + that.lower, this.upper + that.upper)
203     else if(that.upper.rhs > this.lower.lhs && that.lower.lhs < this.upper.rhs)
204       new IntervalImpl[T](this.lower + that.lower, this.upper + that.upper)
205     else if(this.upper.rhs == that.lower.lhs && (this.upper.include || that.lower.include))
206       new IntervalImpl[T](this.lower + that.lower, this.upper + that.upper)
207     else if(that.upper.rhs == this.lower.lhs && (that.upper.include || this.lower.include))
208       new IntervalImpl[T](this.lower + that.lower, this.upper + that.upper)
209     else
210       throw new IllegalArgumentException(s"The intervals $this and $that don't touch.")
211   }
212 
213   /**complement ; all elements which are members of this but not members of rhs*/
214   def -(that: IntervalImpl[T]) =
215     new IntervalImpl[T](this.lower * (-that.upper), this.upper * (-that.lower))
216 
217   /**intersection */
218   def *(that: IntervalImpl[T]) =
219     if(this.upper.rhs >= that.lower.lhs && this.lower.lhs <= that.upper.rhs)
220       new IntervalImpl[T](this.lower*that.lower, this.upper*that.upper)
221     else if(that.upper.rhs >= this.lower.lhs && that.lower.lhs <= this.upper.rhs)
222       new IntervalImpl[T](this.lower*that.lower, this.upper*that.upper)
223     else
224       throw new IllegalArgumentException(s"The intervals $this and $that are disjunct.")
225
