44     override def queryFrom(temporal: TemporalAccessor): Temporal =
45     {
46       TemporalQueries.localDate().queryFrom(temporal)
47         .`with`(TemporalAdjusters.nextOrSame(weekDayPredicate.weekDay))
48         .plusWeeks(count-1)
49     }
50   }
51   def backward(count:Int) = new TemporalQuery[Temporal] {

56         .minusWeeks(count-1)
57     }
58   }
59   override def adjustInto(anchor: Temporal): Temporal = {
60     (number match {
61       case i if(i > 0) => forward(i)
