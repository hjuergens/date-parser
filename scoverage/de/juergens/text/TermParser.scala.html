21   def standardPeriod: Parser[Period] = wholeNumber ~ timeUnit ^^ {
22     case number ~ ChronoUnit.DAYS => Period.ofDays(number.toInt)
23     case number ~ ChronoUnit.WEEKS => Period.ofWeeks(number.toInt)
24     case number ~ ChronoUnit.MONTHS => Period.ofMonths(number.toInt)
25     case number ~ ChronoUnit.YEARS => Period.ofYears(number.toInt)
26   }
27 
28   private def infinityPeriod: Parser[Period] = "Infinity" ^^
29     { _ => Period.ofYears(Integer.MAX_VALUE) }
30 
31   private def lPeriod: Parser[Period] = "L" ^^
32     { _ => Period.ofWeeks(3) }
33 
34   private def periodSum: Parser[Period] = (standardPeriod *) ^^
35     { case periods => periods.foldRight(Period.ZERO)((p, s) => p plus s) }
36 
37   def period: Parser[Period] = infinityPeriod | lPeriod | periodSum
38 
39   def periodAdjuster: Parser[BusinessDayConvention => LocalDate => LocalDate]
40   = ("+" | "-") ~ period ^^
41     {
42       case "+" ~ period =>
43         (bdc:BusinessDayConvention) => (date: LocalDate) =>bdc(date.plus(period))
44       case "-" ~ period =>
45         (bdc:BusinessDayConvention) => (date: LocalDate) =>bdc(date.minus(period))
46     }

49   def timeUnit: Parser[ChronoUnit] = ("D" | "W" | "M" | "Y") ^^
50     {
51       case "D" => ChronoUnit.DAYS
52       case "W" => ChronoUnit.WEEKS
53       case "M" => ChronoUnit.MONTHS
54       case "Y" => ChronoUnit.YEARS
55     }

57   def date: Parser[LocalDate] = year~"-"~month~"-"~dayOfMonth ^^
58     { case year~"-"~month~"-"~dayOfMonth => LocalDate.of(year.getValue,month,dayOfMonth.getValue) }

103   private def thirdWednesday(y:Int,m:Month) =
104     YearMonth.of(y,m).atDay(1)
105       .`with`(TemporalAdjusters.nextOrSame(DayOfWeek.WEDNESDAY))
106       .plus(2, ChronoUnit.WEEKS)

108   /** international money market */
109   def imm: Parser[LocalDate => LocalDate] = deliveryMonth ~ wholeNumber ^^

111       case dm ~ y => {
112         (anchor) =>
113           val yr = (anchor.getYear / 10) * 10 +y.toInt
114           thirdWednesday(yr, dm)
115       }
