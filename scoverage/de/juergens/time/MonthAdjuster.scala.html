8 case class MonthAdjuster(ordinal: Ordinal, month: java.time.Month, direction: Direction)

16   val isMonth = (accessor: TemporalAccessor) => accessor.get(ChronoField.MONTH_OF_YEAR) == month.getValue
17 
18   override def adjustInto(temporal: Temporal): Temporal = {
19     var date = temporal
20 
21     require(temporal.isSupported(ChronoField.MONTH_OF_YEAR), s"$temporal has to support month-of-year")
22     temporal.ensuring(_.isSupported(ChronoField.MONTH_OF_YEAR))
23 
24     if (number > 0)
25       date = date.`with`(ChronoField.MONTH_OF_YEAR, month.getValue)
26 
27     val rest: Int = {
28       if (isMonth(date)) {
29         signum * number
30       } else {
31         if (direction == Up && isBeforeMonth(date)) signum * number - 1
32         else if (direction == Down && isAfterMonth(date)) signum * number + 1
33         else signum * number
34       }
35     }
36 
37     val currentYear = date.get(ChronoField.YEAR)
38     date = date.`with`(ChronoField.YEAR, currentYear + rest)
39 
40     direction match {
41       case Up => date = TemporalAdjusters.lastDayOfMonth().adjustInto(date)
42       case Down => date = TemporalAdjusters.firstDayOfMonth().adjustInto(date)
43     }
44 
45     assert( temporal.get(ChronoField.MONTH_OF_YEAR) == month.getValue )
46

48   }
