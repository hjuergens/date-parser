96      private static String[] getAnnotatedClasses(Class<?> klass) throws InitializationError {
97          SuiteWithParameters.SuiteDirectories annotation = klass.getAnnotation(SuiteWithParameters.SuiteDirectories.class);
98          if(annotation == null) {
99              throw new InitializationError(String.format("class \'%s\' must have a SuiteDirectories annotation", new Object[]{klass.getName()}));
100         } else {
101             return annotation.value();
102         }

112         public ParameterizedFromFile(Class<?> klass, File file) throws Throwable {
113             super(klass, NO_RUNNERS);
114             fFile = file;
115             ParametersRunnerFactory runnerFactory = this.getParametersRunnerFactory(klass);
116             Parameterized.Parameters parameters = (Parameterized.Parameters)this.getParametersMethod().getAnnotation(Parameterized.Parameters.class);
117             this.runners = Collections.unmodifiableList(this.createRunnersForParameters(this.allParameters(), parameters.name(), runnerFactory));

120         private ParametersRunnerFactory getParametersRunnerFactory(Class<?> klass) throws InstantiationException, IllegalAccessException {
121             Parameterized.UseParametersRunnerFactory annotation = (Parameterized.UseParametersRunnerFactory)klass.getAnnotation(Parameterized.UseParametersRunnerFactory.class);
122             if(annotation == null) {
123                 return DEFAULT_FACTORY;
124             } else {
125                 Class factoryClass = annotation.value();
126                 return (ParametersRunnerFactory)factoryClass.newInstance();
127             }
128         }

159         private FrameworkMethod getParametersMethod() throws Exception {
160             List methods = this.getTestClass().getAnnotatedMethods(Parameterized.Parameters.class);
161             Iterator i$ = methods.iterator();
162 
163             FrameworkMethod each;
164             do {
165                 if(!i$.hasNext()) {
166                     throw new Exception("No public static parameters method on class " + this.getTestClass().getName());
167                 }
168 
169                 each = (FrameworkMethod)i$.next();
170             } while(!each.isStatic() || !each.isPublic());
171 
172             return each;
173         }

175         private List<Runner> createRunnersForParameters(Iterable<Object> allParameters, String namePattern, ParametersRunnerFactory runnerFactory) throws InitializationError, Exception {
176             try {
177                 List e = this.createTestsForParameters(allParameters, namePattern);
178                 ArrayList runners = new ArrayList();
179                 Iterator i$ = e.iterator();
180 
181                 while(i$.hasNext()) {
182                     TestWithParameters test = (TestWithParameters)i$.next();
183                     runners.add(runnerFactory.createRunnerForTestWithParameters(test));
184                 }
185 
186                 return runners;
187             } catch (ClassCastException var8) {
188                 throw this.parametersMethodReturnedWrongType();
189             }
190         }

192         private List<TestWithParameters> createTestsForParameters(Iterable<Object> allParameters, String namePattern) throws Exception {
193             int i = 0;
194             ArrayList children = new ArrayList();
195             Iterator i$ = allParameters.iterator();
196 
197             while(i$.hasNext()) {
198                 Object parametersOfSingleTest = i$.next();
199                 children.add(this.createTestWithNotNormalizedParameters(namePattern, i++, parametersOfSingleTest));
200             }
201 
202             return children;
203         }

205         private Exception parametersMethodReturnedWrongType() throws Exception {
206             String className = this.getTestClass().getName();
207             String methodName = this.getParametersMethod().getName();
208             String message = MessageFormat.format("{0}.{1}() must return an Iterable of arrays.", new Object[]{className, methodName});
209             return new Exception(message);
210         }

212         private static TestWithParameters createTestWithParameters(TestClass testClass, String pattern, int index, Object[] parameters) {
213             String finalPattern = pattern.replaceAll("\\{index\\}", Integer.toString(index));
214             String name = MessageFormat.format(finalPattern, parameters);
215             return new TestWithParameters("[" + name + "]", testClass, Arrays.asList(parameters));
216         }
