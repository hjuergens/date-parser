619         while (true) {
620     
621           if (zzCurrentPosL < zzEndReadL)
622             zzInput = zzBufferL[zzCurrentPosL++];
623           else if (zzAtEOF) {
624             zzInput = YYEOF;

626           }
627           else {
628             // store back cached positions
629             zzCurrentPos  = zzCurrentPosL;
630             zzMarkedPos   = zzMarkedPosL;
631             boolean eof = zzRefill();
632             // get translated positions and possibly new buffer

634             zzMarkedPosL   = zzMarkedPos;

636             zzEndReadL     = zzEndRead;

642               zzInput = zzBufferL[zzCurrentPosL++];

644           }
645           int zzNext = zzTransL[ zzRowMapL[zzState] + zzCMapL[zzInput] ];
646           if (zzNext == -1) break zzForAction;

648

650           if ( (zzAttributes & 1) == 1 ) {
651             zzAction = zzState;
652             zzMarkedPosL = zzCurrentPosL;
653             if ( (zzAttributes & 8) == 8 ) break zzForAction;

656         }
