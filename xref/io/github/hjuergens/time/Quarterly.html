4   import org.slf4j.Logger;
5   import org.slf4j.LoggerFactory;

7   import java.util.Iterator;

9   /**
10   * TODO migrate to org.threeten.extra.Quarter
11   * The calendar year can be divided into four quarters, often abbreviated as Q1, Q2, Q3, and Q4.
12   * First quarter, Q1: 1 January–31 March (90 days or 91 days in leap years)
13   * Second quarter, Q2: 1 April-30 June (91 days)
14   * Third quarter, Q3: 1 July-30 September (92 days* )
15   * Fourth quarter, Q4: 1 October-31 December (92 days)
16   * When combined with a year, the quarter/year combination is abbreviated as QxYYYY (Q42016) or
17   * the slightly less confusing xQYYYY (4Q2016)
18   * [https://en.wikipedia.org/wiki/Calendar_year]
19   */
20  public class Quarterly implements Iterator<DateTime> {
21      final private Logger logger = LoggerFactory.getLogger(this.getClass());
22  
23      volatile private DateTime current = null;
24      private final int scalar;
25  
26      Quarterly(DateTime current, int scalar) {
27          this.current = current.withTime(0,0,0,0);
28          this.scalar = scalar;
29          logger.trace(current.toString() + ";" + scalar);
30      }
31      @Override
32      public boolean hasNext() {
33          logger.trace(current.toString() + ";" + scalar);
34          return true;
35      }
36  
37      @Override
38      public DateTime next() {
39          final int addMonth = 3 - (current.getMonthOfYear()-1) % 3;
40          logger.trace("addMonth=" + addMonth);
41          logger.trace("before=" + current);
42          current = current.plusMonths(addMonth).dayOfMonth().setCopy(1); // next month first day
43          logger.trace("after=" + current);
44          return current;
45      }
46  
47      @Override
48      public void remove() {
49          logger.trace("remove");
50          throw new java.lang.UnsupportedOperationException();
51      }
52  }
