3   public class Ring {
4       final private int m;
5   
6       public Ring(int m) {
7           this.m = m;
8       }
9   
10      public int minus(int lhs, int rhs) {
11          if(lhs <= 0 || lhs > m) throw new IllegalArgumentException("Left hand side out of range.");
12          if(rhs <= 0 || rhs > m) throw new IllegalArgumentException("Right hand side out of range.");
13          return lhs-rhs < 0 ? (lhs-rhs)+m : lhs-rhs;
14      }
15  
16      public int plus(int lhs, int amount) {
17          if(lhs <= 0 || lhs > m) throw new IllegalArgumentException("Left hand side out of range.");
18          //if(rhs <= 0 || rhs > m) throw new IllegalArgumentException("Right hand side out of range.");
19  
20          return Math.floorMod(lhs+amount-1, m)+1;
21      }
